### data base UNIVERSITA'

 ## TABELLE

  # DIPARTIMENTO
   -id PRIMARY_KEY AUTO_INCREMENT NOT NULL INT
   -nome dipartimento VARCHAR(20) NOT NULL

  # CORSI_DI_LAUREA
  -id PRIMARY_KEY AUTO_INCREMENT NOT NULL UNIQUE
  -nome VARCHAR(20) NOT NULL
  -dipartimento_id INT NOT NULL

  # CORSO
  -id PRIMARY_KEY AUTO_INCREMENT NOT NULL UNIQUE
  -nome VARCHAR(20)
  -corso_di_laurea_id INT NOT NULL

  # STUDENTE
  -id PRIMARY_KEY AUTO_INCREMENT NOT NULL UNIQUE
  -nome VARCHAR(20) NOT NULL
  -cognome VARCHAR(20) NOT NULL
  -codice matricola VARCHAR(9) NOT NULL UNIQUE
  -corso_di_laurea_id FK INT NOT NULL

  # INSEGNANTE
  -id PRIMARY_KEY AUTO_INCREMENT NOT NULL UNIQUE
  -nome VARCHAR(20) NOT NULL
  -cognome VARCHAR(20) NOT NULL
  -corso_id INT NOT NULL

  # CORSO INSEGNANTE
  -id PRIMARY_KEY AUTO_INCREMENT NOT NULL UNIQUE
  -corso_id INT NOT NULL
  -insegnante_id INT NOT NULL

  # APPELLO ESAME
  -id PRIMARY_KEY AUTO_INCREMENT NOT NULL UNIQUE
  -corso_id FK INT NOT NULL

  # ESITO ESAME
  -id PRIMARY_KEY AUTO_INCREMENT NOT NULL UNIQUE
  -studente_id FK INT NOT NULL
  -appello_id FK INT NOT NULL
  -esito TINYINT NOT NULL
